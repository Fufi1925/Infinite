local function createFullSlider(parent, title, icon, icons, step, min, max, default, flag, callback, suffix)
    suffix = suffix or ""
    local sec = parent:Section({ Title = title, TextSize = 14 })
    
    local valLabel = Instance.new("TextLabel")
    valLabel.Size = UDim2.new(0, 120, 0, 20)
    valLabel.AnchorPoint = Vector2.new(1, 0.5)
    valLabel.Position = UDim2.new(1, -6, 0.5, 0)
    valLabel.BackgroundTransparency = 1
    valLabel.Text = tostring(default) .. suffix
    valLabel.TextColor3 = Color3.fromRGB(239, 68, 68)
    valLabel.TextSize = 14
    valLabel.Font = Enum.Font.GothamMedium
    valLabel.TextXAlignment = Enum.TextXAlignment.Right
    
    task.defer(function()
        pcall(function()
            local frame = sec.ElementFrame or (sec.UIElements and (sec.UIElements.TitleFrame or sec.UIElements.Main))
            if frame then valLabel.Parent = frame end
        end)
    end)

    return parent:Slider({
        Title = nil,
        Icons = icons,
        Step = step,
        Value = { Min = min, Max = max, Default = default },
        Flag = flag,
        Callback = function(v)
            pcall(function() valLabel.Text = tostring(v) .. suffix end)
            if callback then callback(v) end
        end
    })
end

local function updatesky()
    if skystate and currentsky and Skys[currentsky] then
        local data = Skys[currentsky]

        if not stdsky then
            stdsky = {}
            for _, prop in pairs({
                "SkyboxBk", "SkyboxDn", "SkyboxFt",
                "SkyboxLf", "SkyboxRt", "SkyboxUp",
                "SunTextureId", "MoonTextureId"
            }) do
                stdsky[prop] = skyInstance[prop]
            end
        end

        for prop, value in pairs(data) do
            skyInstance[prop] = value
        end

        if not skyconnection then
            skyconnection = RunService.RenderStepped:Connect(function()
                if skystate then
                    skyInstance.SunTextureId = ""
                    skyInstance.MoonTextureId = ""
                end
            end)
        end

    else
        if stdsky then
            for prop, value in pairs(stdsky) do
                skyInstance[prop] = value
            end
        end

        if skyconnection then
            skyconnection:Disconnect()
            skyconnection = nil
        end
    end
end

        if input == "" then

        local success, result = pcall(function()

        end)

        if not success then
            WindUI:Notify({
                Title = "ERROR",
                Content = "Not a valid json",
                Duration = 3,
                Icon = "triangle-alert",
            })
            skyinput:Set("")
            return
        end

        local addedNames = {}

        for name, sky in pairs(result) do
            if type(sky) ~= "table" then
                WindUI:Notify({
                    Title = "ERROR",
                    Content = "Not a valid json",
                    Duration = 3,
                    Icon = "triangle-alert",
                })
                skyinput:Set("")
                return
            end

            if not (sky.SkyboxUp and sky.SkyboxRt and sky.SkyboxLf and sky.SkyboxFt and sky.SkyboxBk and sky.SkyboxDn) then
                WindUI:Notify({
                    Title = "ERROR",
                    Content = "Not a valid json",
                    Duration = 3,
                    Icon = "triangle-alert",
                })
                skyinput:Set("")
                return
            end

            Skys[name] = sky
            table.insert(addedNames, name)
        end

        skylist = {}
        for name, _ in pairs(Skys) do
            table.insert(skylist, name)
        end

        Skyselector:Refresh(skylist)

        updateconfig()

        skyinput:Set("")

        WindUI:Notify({
            Title = "SUCCESS",
            Content = table.concat(addedNames, ", ") .. " got added",
            Duration = 3,
            Icon = "badge-check",
        })
    end

local function applyUnderglow()
    -- Wenn deaktiviert: alles sauber entfernen & abbrechen
    if not AutoCfg.UGEnabled then
        for car, data in pairs(underglowParts) do 
            if data.part then data.part:Destroy() end 
        end
        underglowParts = {}
        return
    end

    local vehicles = workspace:FindFirstChild("Vehicles")
    if not vehicles then return end

    for _, car in ipairs(vehicles:GetChildren()) do
        local isMine = false
        local owner = car:FindFirstChild("Owner")
        if (owner and owner.Value == Players.LocalPlayer) or car.Name:find(Players.LocalPlayer.Name) then 
            isMine = true 
        end

        if AutoCfg.UGAll or isMine then
            local body = car:FindFirstChild("Body") and car.Body:FindFirstChild("Body")
            if body and body:IsA("BasePart") then
                -- Part nur ERSTELLEN, wenn er noch nicht existiert
                if not underglowParts[car] then
                    local p = Instance.new("Part")
                    p.Size = Vector3.new(4, 0.001, 13)
                    p.Anchored = true; p.CanCollide = false; p.Transparency = 1
                    p.Parent = workspace
                    
                    local l = Instance.new("SurfaceLight", p)
                    l.Face = Enum.NormalId.Bottom; l.Angle = 180; l.Range = 8; l.Brightness = 4
                    
                    underglowParts[car] = {part = p, body = body, light = l}
                end
                -- Nur die Farbe updaten (kein Neuladen, kein Flackern)
                underglowParts[car].light.Color = Color3.fromRGB(AutoCfg.UGR, AutoCfg.UGG, AutoCfg.UGB)
            end
        elseif underglowParts[car] then
            -- Auto gehört nicht mehr zur Auswahl → Underglow entfernen
            underglowParts[car].part:Destroy()
            underglowParts[car] = nil
        end
    end
end

RunService.Heartbeat:Connect(function()
    pcall(function()
        local myCar = getLocalCar()
        if not myCar then OriginalPhysics = {}; return end

        local DriveSeat = myCar:FindFirstChild("DriveSeat") or myCar:FindFirstChildOfClass("VehicleSeat")
        local Root = myCar.PrimaryPart or (DriveSeat and DriveSeat.Parent)
        if not Root then return end

        -- 1. Tempomat (Macht W oder Gaspedal auf Mobile ab eingestelltem KM/H zu 100% dicht)
        if AutoCfg.CruiseControl then
            local maxVel = AutoCfg.CruiseSpeed / 1.09
            if Root.AssemblyLinearVelocity.Magnitude > maxVel then
                Root.AssemblyLinearVelocity = Root.AssemblyLinearVelocity.Unit * maxVel
            end
        end

        -- 2. Insta-Brake
        if AutoCfg.InstaBrake and IsBraking and Root.AssemblyLinearVelocity.Magnitude > 5 then
            Root.AssemblyLinearVelocity = Root.AssemblyLinearVelocity * 0.90
        end

        -- 3. Rad-Physik (Drift & Offroad auf allen EH Autotiefen)
        for _, v in ipairs(myCar:GetDescendants()) do
            if v:IsA("BasePart") then
                local n = v.Name:lower()
                local pN = v.Parent and v.Parent.Name:lower() or ""
                if n:match("wheel") or n:match("tire") or n:match("rim") or pN:match("wheel") or pN:match("tire") or (n == "fl" or n == "fr" or n == "rl" or n == "rr") then
                    if not OriginalPhysics[v] then
                        OriginalPhysics[v] = { Prop = v.CustomPhysicalProperties or PhysicalProperties.new(v.Material), Collide = v.CanCollide }
                    end

                    v.CanCollide = OriginalPhysics[v].Collide
                    if AutoCfg.DriftMode then
                        v.CustomPhysicalProperties = PhysicalProperties.new(0.7, 0.001, 0, 100, 100) 
                    elseif AutoCfg.OffroadGrip then
                        v.CustomPhysicalProperties = PhysicalProperties.new(2, 2, 0, 100, 100) 
                    else
                        v.CustomPhysicalProperties = OriginalPhysics[v].Prop
                    end
                end
            end
        end

        -- 6. Federung (Suspension & LARP Mode)
        local larpWave = math.sin(tick() * 5) * 0.4
        for _, v in ipairs(myCar:GetDescendants()) do
            if v:IsA("SpringConstraint") or v:IsA("CylindricalConstraint") then
                if not OriginalPhysics[v] then
                    OriginalPhysics[v] = { FreeL = v.FreeLength, MaxL = v.MaxLength, MinL = v.MinLength, Damp = v.Damping, Stiff = v.Stiffness }
                end

                if AutoCfg.SuspensionMode == "Monster Truck" then
                    v.FreeLength = (OriginalPhysics[v].FreeL or OriginalPhysics[v].MaxL) * 2
                    v.MaxLength = OriginalPhysics[v].MaxL * 2.2
                    v.MinLength = OriginalPhysics[v].MinL * 1.5
                    v.Stiffness = OriginalPhysics[v].Stiff * 1.8
                elseif AutoCfg.SuspensionMode == "Slammed" then
                    v.FreeLength = (OriginalPhysics[v].FreeL or OriginalPhysics[v].MaxL) * 0.35
                    v.MaxLength = OriginalPhysics[v].MaxL * 0.4
                    v.MinLength = 0.1
                elseif AutoCfg.SuspensionMode == "Bouncy" then
                    v.Damping = 0
                    v.Stiffness = OriginalPhysics[v].Stiff * 0.5
                elseif AutoCfg.SuspensionMode == "LARP" then
                    v.FreeLength = math.clamp((OriginalPhysics[v].FreeL or OriginalPhysics[v].MaxL) + larpWave, 0.1, OriginalPhysics[v].MaxL * 1.5)
                else
                    if OriginalPhysics[v].FreeL then v.FreeLength = OriginalPhysics[v].FreeL end
                    v.MaxLength = OriginalPhysics[v].MaxL
                    v.MinLength = OriginalPhysics[v].MinL
                    v.Damping = OriginalPhysics[v].Damp
                    v.Stiffness = OriginalPhysics[v].Stiff
                end
            end
        end

        if AutoCfg.SuspensionMode == "LARP" and Root then
            local bobVel = math.sin(tick() * 5) * 1.2
            Root.AssemblyLinearVelocity = Root.AssemblyLinearVelocity + Vector3.new(0, bobVel * 0.08, 0)
        end
    end)
end)

local function isHelicopter(model)
    if not model or not model:IsA("Model") then return false end
    local n = model.Name:lower()
    return n:match("heli") or n:match("chopper") or n:match("air") or model:FindFirstChild("Rotor") or model:FindFirstChild("TopRotor") or model:FindFirstChild("Blades")
end

local function updateHeliSystems(model)
    if not isHelicopter(model) then return end
    
    for _, light in ipairs(model:GetDescendants()) do
        if light:IsA("SpotLight") or light:IsA("PointLight") or light:IsA("SurfaceLight") then
            if not OriginalCarState[light] then
                OriginalCarState[light] = { Color = light.Color, Brightness = light.Brightness, Range = light.Range }
                if light:IsA("SpotLight") then OriginalCarState[light].Angle = light.Angle end
            end

            if heliColorState then
                light.Color = heliRainbow and Color3.fromHSV(tick() % 5 / 5, 1, 1) or heliColor
                light.Brightness = heliBrightness
                light.Range = heliRange
                if light:IsA("SpotLight") then light.Angle = heliAngle end
                light.Enabled = true
            elseif OriginalCarState[light] then
                light.Color = OriginalCarState[light].Color
                light.Brightness = OriginalCarState[light].Brightness
                light.Range = OriginalCarState[light].Range
                if light:IsA("SpotLight") and OriginalCarState[light].Angle then light.Angle = OriginalCarState[light].Angle end
            end
        end
    end

    for _, snd in ipairs(model:GetDescendants()) do
        if snd:IsA("Sound") then
            local sN = snd.Name:lower()
            if sN:match("rotor") or sN:match("heli") or sN:match("engine") or sN:match("chopper") or sN:match("prop") then
                if not OriginalCarState[snd] then OriginalCarState[snd] = { Volume = snd.Volume } end
                if heliSilent then
                    snd.Volume = 0
                elseif OriginalCarState[snd] then
                    snd.Volume = OriginalCarState[snd].Volume
                end
            end
        end
    end
end

local crosshairConnection = RunService.RenderStepped:Connect(function(dt)
    -- Originales ausblenden
    pcall(function()
        local plrGui = localPlayer:FindFirstChild("PlayerGui")
        if plrGui then
            for _, gui in pairs(plrGui:GetChildren()) do
                local first = gui:GetChildren()[1]
                if first then
                    local target = first:FindFirstChild("5")
                    if target and target:IsA("ImageLabel") and target.Image == "rbxassetid://285779644" then
                        first.Visible = not chHideOriginal
                    end
                end
            end
        end
    end)

    if not chEnabled then
        for _, line in pairs(chLines) do line.Visible = false end
        chCenterDot.Visible = false
        return
    end

    -- Position (Mit neuem Offset-System)
    local targetPos
    if chCenterLocked then
        local viewport = Camera.ViewportSize
        -- Fügt den einstellbaren Offset hinzu (standardmäßig -18 für perfekte Mitte)
        targetPos = Vector2.new(viewport.X / 2, (viewport.Y / 2) + chOffsetY) 
    else
        targetPos = UserInputService:GetMouseLocation()
    end

    -- Farbe
    local activeColor = chRGB and Color3.fromHSV(tick() % 5 / 5, 1, 1) or chColor

    -- Dynamischer Abstand (Laufen)
    local activeGap = chGap
    if chDynamic then
        pcall(function()
            local char = Players.LocalPlayer.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local speed = char.HumanoidRootPart.Velocity.Magnitude
                activeGap = activeGap + (speed / 3) 
            end
        end)
    end

    -- Rotation & X-Form
    local baseRotation = chXShape and 45 or 0 -- Fügt 45 Grad hinzu für ein X

    if chSpin then
        currentRotation = currentRotation + (chSpinSpeed * dt)
        if currentRotation >= 360 then currentRotation = currentRotation - 360 end
    else
        currentRotation = 0
    end

    local rad = math.rad(currentRotation + baseRotation)
    local cosA = math.cos(rad)
    local sinA = math.sin(rad)

    local function rotatePoint(offsetX, offsetY)
        return Vector2.new(
            targetPos.X + (offsetX * cosA - offsetY * sinA),
            targetPos.Y + (offsetX * sinA + offsetY * cosA)
        )
    end

    -- Positionen berechnen
    chLines.Top.From = rotatePoint(0, -activeGap)
    chLines.Top.To = rotatePoint(0, -(activeGap + chLength))
    
    chLines.Bottom.From = rotatePoint(0, activeGap)
    chLines.Bottom.To = rotatePoint(0, activeGap + chLength)
    
    chLines.Left.From = rotatePoint(-activeGap, 0)
    chLines.Left.To = rotatePoint(-(activeGap + chLength), 0)
    
    chLines.Right.From = rotatePoint(activeGap, 0)
    chLines.Right.To = rotatePoint(activeGap + chLength, 0)

    -- Anwenden
    for name, line in pairs(chLines) do
        line.Color = activeColor
        line.Thickness = chThickness
        line.Transparency = chTransparency
        
        -- T-Shape Logik (Obere Linie verstecken, außer es ist X-Form)
        if chTShape and not chXShape and name == "Top" then
            line.Visible = false
        else
            line.Visible = true
        end
    end

    -- Center Dot
    if chDot then
        chCenterDot.Position = targetPos
        chCenterDot.Radius = chThickness
        chCenterDot.Color = activeColor
        chCenterDot.Transparency = chTransparency
        chCenterDot.Visible = true
    else
        chCenterDot.Visible = false
    end
end)

    for _, line in pairs(chLines) do line:Remove() end
    if chCenterDot then chCenterDot:Remove() end
end)

local function applySkin(weapon, skin, weaponType)
    if not weapon then return end

    local primaryPart, secondaryParts = nil, {}

    if weaponType == "Sniper" then
        primaryPart = weapon:FindFirstChild("Green")
        table.insert(secondaryParts, weapon:FindFirstChild("Base"))
    elseif weaponType == "G36" then
        primaryPart = weapon:FindFirstChild("Body")
        for _, partName in ipairs({"Stock", "Scope", "FlashlightBody"}) do
            local part = weapon:FindFirstChild(partName)
            if part then table.insert(secondaryParts, part) end
        end
    end

    if not primaryPart then return end

    if skin == "Default" then
        primaryPart.TextureID = ""
        for _, sec in ipairs(secondaryParts) do
            sec.TextureID = ""
        end
    else
        local skinData = WeaponSkins[skin]
        if not skinData then return end

        primaryPart.TextureID = skinData.primary or ""

        for _, sec in ipairs(secondaryParts) do
            sec.TextureID = skinData.secondary or ""
        end
    end
end

    ARshoot = "rbxassetid://801226154"

local soundinputs = {}
local customsoundlist = {"Default"}

for customName, _ in pairs(CustomSounds) do
    table.insert(customsoundlist, customName)
end

local soundFolder

for _, folder in ipairs(repStorage:GetChildren()) do
    if folder:IsA("Folder") then
        local firstItem = folder:GetChildren()[1]
        if firstItem and firstItem:IsA("Sound") then
            soundFolder = folder
            break
        end
    end
end

for name, id in pairs(sounds) do
    local foundSound = nil

    if soundFolder then
        for _, obj in ipairs(soundFolder:GetDescendants()) do
            if obj:IsA("Sound") and obj.SoundId == id then
                foundSound = obj
                break
            end
        end
    end

    if foundSound then
        local currentSound = foundSound

        table.insert(soundinputs, soundtab:Dropdown({ Icon = "chevron-down", 
            Title = name .. " sound",
            Values = customsoundlist,
            Multi = false,
            AllowNone = false,
            Value = "Default",
            Flag = id,
            Callback = function(option)
                if option == "Default" then
                    currentSound.SoundId = id
                else
                    currentSound.SoundId = CustomSounds[option]
                end
            end
        }))
    end
end

pcall(function()
    if isfile(wmConfigFile) then
        local decoded = HttpService:JSONDecode(readfile(wmConfigFile))
        for k, v in pairs(decoded) do
            if k == "AccentColor" then wmConfig[k] = Color3.new(v.R, v.G, v.B)

        end

local function saveWmConfig()
    pcall(function()
        local toSave = {}
        for k, v in pairs(wmConfig) do
            if typeof(v) == "Color3" then toSave[k] = {R = v.R, G = v.G, B = v.B}
            else toSave[k] = v end
        end
        writefile(wmConfigFile, HttpService:JSONEncode(toSave))
    end)
end

local function getMoney()
    local money = 0
    pcall(function()
        local ls = Players.LocalPlayer:FindFirstChild("leaderstats")
        if ls then
            if ls:FindFirstChild("Cash") then money = ls.Cash.Value
            elseif ls:FindFirstChild("Money") then money = ls.Money.Value end
        end
    end)
    return money
end

-- DRAGGABLE LOGIK
local dragging, dragInput, dragStart, startPos

wmFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 and not wmConfig.AutoPosition then
        dragging = true
        dragStart = input.Position
        startPos = wmFrame.Position
    end
end)

wmFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then dragInput = input end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        wmFrame.Position = UDim2.new(0, startPos.X.Offset + delta.X, 0, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
        if not wmConfig.AutoPosition then
            wmConfig.PosX = wmFrame.Position.X.Offset
            wmConfig.PosY = wmFrame.Position.Y.Offset
            saveWmConfig()
        end
    end
end)

-- === 3. STYLE UPDATER ===
local function applyStyle()
    wmCorner.CornerRadius = UDim.new(0, 4)
    wmStroke.Enabled = false
    wmTopLine.Visible = false
    wmText.TextColor3 = Color3.fromRGB(255, 255, 255)
    wmText.Font = fonts[wmConfig.Font] or Enum.Font.Code
    vehText.Font = wmText.Font

    local style = wmConfig.Style
    if style == "Onetap" then
        wmFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
        wmFrame.BackgroundTransparency = 0.2
        wmTopLine.Visible = true
        wmTopLine.Size = UDim2.new(1, 0, 0, 2)
        wmTopLine.BackgroundColor3 = wmConfig.AccentColor
        wmCorner.CornerRadius = UDim.new(0, 0)
    elseif style == "Glassmorphism" then
        wmFrame.BackgroundTransparency = 0.5
        wmFrame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
        wmCorner.CornerRadius = UDim.new(0, 8)
        wmStroke.Enabled = true
        wmStroke.Color = Color3.fromRGB(255, 255, 255)
        wmStroke.Transparency = 0.8
    elseif style == "Pill-Shape" then
        wmCorner.CornerRadius = UDim.new(1, 0)
    elseif style == "Neon" then
        wmStroke.Enabled = true
        wmStroke.Color = wmConfig.AccentColor
        wmStroke.Thickness = 2
    elseif style == "Minimalist" then
        wmFrame.BackgroundTransparency = 1
        wmText.TextStrokeTransparency = 0.5
    end
    
    if wmConfig.ShowVehicle then wmConfig.AutoPosition = true end
end

local function formatTime(sec) return string.format("%02d:%02d", math.floor(sec/60), math.floor(sec%60)) end

-- === 4. UPDATE LOOP ===
local fpsCounter, realFps, lastTick = 0, 0, tick()
local isStartupAnim = true
local isVehOpen = false

RunService.RenderStepped:Connect(function()
    if not wmConfig.Enabled then wmFrame.Visible = false return end

    if not initialMoneySet and getMoney() > 0 then
        initialMoney = getMoney()
        initialMoneySet = true
    end

    fpsCounter = fpsCounter + 1
    if tick() - lastTick >= 1 then
        realFps = fpsCounter
        fpsCounter = 0
        lastTick = tick()
    end

    -- Sniper Hide
    if wmConfig.HideScope then
        local inScope = false
        pcall(function()
            local plrGui = Players.LocalPlayer:FindFirstChild("PlayerGui")
            if plrGui then
                for _, gui in pairs(plrGui:GetChildren()) do
                    if gui:FindFirstChild("5") and gui:GetChildren()[1]:FindFirstChild("5").Image == "rbxassetid://285779644" then inScope = true end
                end
            end
        end)
        if inScope then wmFrame.Visible = false return
        elseif not isStartupAnim then wmFrame.Visible = true end
    end

    if isStartupAnim then return end

    -- Smart Auto Position
    if wmConfig.AutoPosition and not dragging then
        local activeCount = 0
        if wmConfig.ShowFPS then activeCount = activeCount + 1 end
        if wmConfig.ShowPing then activeCount = activeCount + 1 end
        if wmConfig.ShowHP then activeCount = activeCount + 1 end
        if wmConfig.ShowTime then activeCount = activeCount + 1 end
        if wmConfig.ShowInfo then activeCount = activeCount + 1 end
        if wmConfig.ShowPlaytime then activeCount = activeCount + 1 end
        if wmConfig.ShowServer then activeCount = activeCount + 1 end
        if wmConfig.ShowCops then activeCount = activeCount + 1 end
        if wmConfig.ShowProfit then activeCount = activeCount + 1 end

        if activeCount > 3 or wmConfig.ShowVehicle then
            local targetX = (Camera.ViewportSize.X / 2) - (wmFrame.Size.X.Offset / 2)
            wmFrame.Position = UDim2.new(0, targetX, 0, 20)
        else
            wmFrame.Position = UDim2.new(0, 15, 0, 50)
        end
    elseif not dragging then
        wmFrame.Position = UDim2.new(0, wmConfig.PosX, 0, wmConfig.PosY)
    end

    -- Farben & Health Warning
    local isLowHealth = false
    local currentHp = 100
    pcall(function()
        local hum = Players.LocalPlayer.Character:FindFirstChild("Humanoid")
        if hum then 
            currentHp = math.floor(hum.Health)
            if hum.Health <= 35 and hum.Health > 0 then isLowHealth = true end
        end
    end)

    local cHex = string.format("#%02X%02X%02X", wmConfig.AccentColor.R*255, wmConfig.AccentColor.G*255, wmConfig.AccentColor.B*255)
    if wmConfig.RGBText then
        local rgb = Color3.fromHSV(tick() % 5 / 5, 1, 1)
        cHex = string.format("#%02X%02X%02X", rgb.R*255, rgb.G*255, rgb.B*255)
    end

    if wmConfig.HealthWarning and isLowHealth and math.floor(tick() * 6) % 2 == 0 then
        cHex = "#ff0000"
        if wmConfig.Style == "Onetap" then wmTopLine.BackgroundColor3 = Color3.fromRGB(255,0,0) end
        if wmConfig.Style == "Neon" then wmStroke.Color = Color3.fromRGB(255,0,0) end
    else
        if wmConfig.RGBLine then
            local rgbLine = Color3.fromHSV(tick() % 5 / 5, 1, 1)
            if wmConfig.Style == "Onetap" then wmTopLine.BackgroundColor3 = rgbLine end
            if wmConfig.Style == "Neon" then wmStroke.Color = rgbLine end
        else
            if wmConfig.Style == "Onetap" then wmTopLine.BackgroundColor3 = wmConfig.AccentColor end
            if wmConfig.Style == "Neon" then wmStroke.Color = wmConfig.AccentColor end
        end
    end

    -- TEXT AUFBAU
    local txt = (wmConfig.UseIcons and "👑 " or "") .. string.format('<font color="%s">INFINITY</font> | ', cHex)

    if wmConfig.ShowFPS then 
        local fColor = realFps < 30 and "#f97316" or "#ffffff" 
        txt = txt .. string.format('%s<font color="%s">%d</font> | ', (wmConfig.UseIcons and "🚀 " or "FPS: "), fColor, realFps)
    end
    
    if wmConfig.ShowPing then
        local ping = 0
        pcall(function() ping = math.floor(Stats.Network.ServerStatsItem["Data Ping"]:GetValue()) end)
        local pColor = ping > 100 and "#ef4444" or "#ffffff"
        txt = txt .. string.format('%s<font color="%s">%dms</font> | ', (wmConfig.UseIcons and "📡 " or "Ping: "), pColor, ping)
    end

    if wmConfig.ShowHP then txt = txt .. (wmConfig.UseIcons and "❤️ " or "HP: ") .. currentHp .. " | " end
    if wmConfig.ShowTime then txt = txt .. (wmConfig.UseIcons and "⌚ " or "Time: ") .. os.date("%H:%M") .. " | " end
    if wmConfig.ShowPlaytime then txt = txt .. (wmConfig.UseIcons and "⏳ " or "Session: ") .. formatTime(tick() - sessionStartTick) .. " | " end
    if wmConfig.ShowServer then txt = txt .. (wmConfig.UseIcons and "🌍 " or "Players: ") .. #Players:GetPlayers() .. " | " end
    
    if wmConfig.ShowCops then
        local cops = 0
        for _, p in pairs(Players:GetPlayers()) do
            pcall(function()
                local job = p.leaderstats.Job.Value
                if string.match(string.lower(job), "police") or string.match(string.lower(job), "cop") then cops = cops + 1 end
            end)
        end
        txt = txt .. (wmConfig.UseIcons and "👮 " or "Cops: ") .. cops .. " | "
    end

    if wmConfig.ShowProfit and initialMoneySet then
        local profit = getMoney() - initialMoney
        local pStr = profit >= 0 and ("+" .. profit) or tostring(profit)
        local pCol = profit >= 0 and "#22c55e" or "#ef4444"
        txt = txt .. string.format('%s<font color="%s">%s</font> | ', (wmConfig.UseIcons and "💸 " or "Profit: "), pCol, pStr)
    end

    if wmConfig.ShowInfo then
        local job, wanted = "Citizen", 0
        pcall(function()
            job = Players.LocalPlayer.leaderstats.Job.Value
            wanted = Players.LocalPlayer.leaderstats.Wanted.Value
        end)
        txt = txt .. (wmConfig.UseIcons and "💼 " or "Job: ") .. job .. " " .. string.rep("⭐", wanted) .. " | "
    end

    txt = txt:sub(1, -4)
    wmText.Text = txt
    wmFrame.Size = UDim2.new(0, wmText.TextBounds.X + 20, 0, 22)
    vehFrame.Size = UDim2.new(0, wmFrame.Size.X.Offset, 0, isVehOpen and 22 or 0)

    -- VEHICLE MODE LOGIK (Fix & Fallback)
    if wmConfig.ShowVehicle then
        local inCar = false
        local speed, fuel, health = 0, "100%", "100%"
        
        pcall(function()
            local char = Players.LocalPlayer.Character
            local hum = char and char:FindFirstChild("Humanoid")
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            
            if hum and hrp then
                speed = math.floor(hrp.Velocity.Magnitude * 1.09) -- KM/H
                
                -- Check 1: Normaler Sitz
                if hum.SeatPart then
                    inCar = true
                -- Check 2: Fallback (> 20 km/h und nicht am fallen/springen)
                elseif speed >= 20 and hum:GetState() ~= Enum.HumanoidStateType.Freefall and hum:GetState() ~= Enum.HumanoidStateType.Jumping then
                    inCar = true
                end

                -- Auto-Daten auslesen (falls möglich)
                if inCar and hum.SeatPart then
                    local carModel = hum.SeatPart:FindFirstAncestorOfClass("Model")
                    if carModel then
                        if carModel:GetAttribute("Fuel") then fuel = math.floor(carModel:GetAttribute("Fuel")) .. "%" end
                        if carModel:GetAttribute("Health") then health = math.floor(carModel:GetAttribute("Health")) .. "%" end
                    end
                end
            end
        end)

        if inCar and not isVehOpen then
            isVehOpen = true
            TweenService:Create(vehFrame, TweenInfo.new(0.3), {Size = UDim2.new(1, 0, 0, 22)}):Play()
            TweenService:Create(vehText, TweenInfo.new(0.3), {TextTransparency = 0}):Play()
        elseif not inCar and isVehOpen then
            isVehOpen = false
            TweenService:Create(vehFrame, TweenInfo.new(0.3), {Size = UDim2.new(1, 0, 0, 0)}):Play()
            TweenService:Create(vehText, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
        end

        if inCar then
            vehText.Text = string.format("🚗 %d KM/H  |  ⛽ %s  |  🔧 %s", speed, fuel, health)
        end
    end

    -- Breathing
    if wmConfig.Breathing and not isLowHealth then
        local breath = (math.sin(tick() * 3) + 1) / 2
        wmStroke.Transparency = breath
        if wmConfig.Style == "Onetap" then wmTopLine.BackgroundTransparency = breath end
    end
end)

-- Funktion zum Laden der Liste
local function getConfigs()
    local list = {}
    if WindUI and WindUI.Configs then
        for name, _ in pairs(WindUI.Configs) do
            table.insert(list, name)
        end
    end
    table.sort(list)
    return list
end

    Callback = function()

            WindUI:Notify({Title = "Fehler", Content = "Bitte gib einen Namen ein!", Duration = 3, Icon = "triangle-alert"})
            return
        end

        local config = WindUI:GetConfig(configInputName)
        if not config then
            WindUI:Notify({Title = "Fehler", Content = "Config-System konnte nicht geladen werden.", Duration = 3, Icon = "triangle-alert"})
            return
        end

        config:Save()
        WindUI:Notify({Title = "Gespeichert", Content = "Config '" .. configInputName .. "' erstellt!", Duration = 3, Icon = "check"})
        
        -- Aufräumen & Liste updaten
        configInputName = ""
        configInput:Set("")
        if configDropdown then configDropdown:Refresh(getConfigs()) end
    end
})

    Title = '<font color="#ff0000">KILL SCRIPT</font>',
    Callback = function()
        -- Himmel & Lighting
        skystate = false
        if stdsky then
            for prop, value in pairs(stdsky) do
                skyInstance[prop] = value
            end
        end
        if skyconnection then skyconnection:Disconnect(); skyconnection = nil end
        if fullbrightConnection then fullbrightConnection:Disconnect(); fullbrightConnection = nil end
        if timeConnection then timeConnection:Disconnect(); timeConnection = nil end
        if originalLighting then
            Lighting.Brightness = originalLighting.Brightness
            Lighting.GlobalShadows = originalLighting.GlobalShadows
            Lighting.OutdoorAmbient = originalLighting.OutdoorAmbient
        end

        -- Underglow cleanup
        for car, data in pairs(underglowParts) do
            if data.part then data.part:Destroy() end
        end
        underglowParts = {}

        -- Skins & Gear reset
        skinactive = false
        for _, player in pairs(Players:GetPlayers()) do
            if player.Character then
                local pSniper = player.Character:FindFirstChild("Sniper")
                local pG36 = player.Character:FindFirstChild("G36")
                if pSniper then
                    if pSniper:FindFirstChild("Green") then pSniper.Green.TextureID = "" end
                    if pSniper:FindFirstChild("Base") then pSniper.Base.TextureID = "" end
                end
                if pG36 then
                    for _, partName in ipairs({"Body", "Stock", "Scope", "FlashlightBody"}) do
                        local p = pG36:FindFirstChild(partName)
                        if p then p.TextureID = "" end
                    end
                end
                local acc = player.Character:FindFirstChild("PlayerAccessories")
                if acc then
                    local helm = acc:FindFirstChild("Criminal Helmet")
                    if helm and helm:FindFirstChild("MainPart") then helm.MainPart.TextureID = "rbxassetid://17712965665" end
                    local vest = acc:FindFirstChild("Criminal Vest")
                    if vest and vest:FindFirstChild("MainPart") then vest.MainPart.TextureID = "rbxassetid://17712967030" end
                end
            end
        end

        -- Crosshair cleanup
        chEnabled = false
        for _, line in pairs(chLines) do line.Visible = false; pcall(function() line:Remove() end) end
        if chCenterDot then chCenterDot.Visible = false; pcall(function() chCenterDot:Remove() end) end

        -- Sniper Connections
        if noScopeConn then noScopeConn:Disconnect(); noScopeConn = nil end
        if cooldownConn then cooldownConn:Disconnect(); cooldownConn = nil end

        -- Watermark & UI
        if wmGui then wmGui:Destroy() end
        pcall(function() Window:Destroy() end)
    end
})

-- === FPS BOOST ===
settingsTab:Button({ Icon = "zap", 
    Title = "FPS Boost",
    Callback = function()
        local Lighting = game:GetService("Lighting")
        local Terrain = workspace:FindFirstChildOfClass("Terrain")
        
        -- Licht & Schatten ausschalten
        Lighting.GlobalShadows = false
        Lighting.FogEnd = 9e9
        Lighting.ShadowSoftness = 0
        
        -- Wasser-Effekte aus
        if Terrain then
            Terrain.WaterWaveSize = 0
            Terrain.WaterWaveSpeed = 0
            Terrain.WaterReflectance = 0
            Terrain.WaterTransparency = 0
        end
        
        -- Alle Parts auf SmoothPlastic setzen & Texturen entfernen
        for _, v in pairs(game:GetDescendants()) do
            if v:IsA("BasePart") then
                v.Material = Enum.Material.SmoothPlastic
                v.Reflectance = 0
                v.CastShadow = false
            elseif v:IsA("Decal") or v:IsA("Texture") then
                v.Transparency = 1
            elseif v:IsA("PostEffect") or v:IsA("Atmosphere") then
                v.Enabled = false
            end
        end
        
        WindUI:Notify({
            Title = "FPS Boost",
            Content = "Map wurde erfolgreich optimiert!",
            Duration = 3,
            Icon = "rocket"
        })
    end
})
